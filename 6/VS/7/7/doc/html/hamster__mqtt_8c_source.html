334 static int make_hash(const char *s1, const char *s2)
335 {
336  const int InitialValue = 5381;
337  const int M = 33;
338 
339  unsigned int hash;
340  int i;
341  char key[2*(HMSTR_MAX_NAME+1)];
342  /* copy both strings into a single buffer */
343  memset(key, 0, sizeof(key));
344  strcpy(&key[0], s1);
345  strcpy(&key[HMSTR_MAX_NAME+1], s2);
346 
347  /* compute a hash value over the buffer's contents */
348  hash = InitialValue;
349  for(i = 0; i < 2*(HMSTR_MAX_NAME+1); ++i)
350  hash = M * hash + key[i];
351 
352  /* make sure always get a >= 0 number */
353  return hash >> 1U;
354 }
