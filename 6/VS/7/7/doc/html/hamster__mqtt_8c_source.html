650 static int make_hash(const char *s1, const char *s2)
651 {
652  const int InitialValue = 5381;
653  const int M = 33;
654 
655  unsigned int hash;
656  int i;
657  char key[2*(HMSTR_MAX_NAME+1)];
658  /* copy both strings into a single buffer */
659  memset(key, 0, sizeof(key));
660  strcpy(&key[0], s1);
661  strcpy(&key[HMSTR_MAX_NAME+1], s2);
662 
663  /* compute a hash value over the buffer's contents */
664  hash = InitialValue;
665  for(i = 0; i < 2*(HMSTR_MAX_NAME+1); ++i)
666  hash = M * hash + key[i];
667 
668  /* make sure always get a >= 0 number */
669  return hash >> 1U;
670 }
