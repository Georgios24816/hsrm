488 static int make_hash(const char *s1, const char *s2)
489 {
490  const int InitialValue = 5381;
491  const int M = 33;
492 
493  unsigned int hash;
494  int i;
495  char key[2*(HMSTR_MAX_NAME+1)];
496  /* copy both strings into a single buffer */
497  memset(key, 0, sizeof(key));
498  strcpy(&key[0], s1);
499  strcpy(&key[HMSTR_MAX_NAME+1], s2);
500 
501  /* compute a hash value over the buffer's contents */
502  hash = InitialValue;
503  for(i = 0; i < 2*(HMSTR_MAX_NAME+1); ++i)
504  hash = M * hash + key[i];
505 
506  /* make sure always get a >= 0 number */
507  return hash >> 1U;
508 }
